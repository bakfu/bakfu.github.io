A NLP/machine learning toolkit.

example

>>> import bakfu
>>> data = [(i,a) for i,a in enumerate((
...            'Test',
...            'Other test',
...            'Something else',
...            'Nevermind'
...            ))]
>>> baf = bakfu.Chain().load('data.simple',data)
>>> baf.process('vectorize.sklearn')
>>> baf.process('cluster.ward',n_clusters=2)
>>> baf.get_chain('result').tolist()
